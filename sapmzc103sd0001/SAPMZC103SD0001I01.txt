*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.

  CASE gv_okcode.
    WHEN 'SAVE'.
      PERFORM set_save.
    WHEN 'SRCH'.
      PERFORM set_search.
    WHEN 'RFSH'.
      PERFORM set_clear.
  ENDCASE.

ENDMODULE.
